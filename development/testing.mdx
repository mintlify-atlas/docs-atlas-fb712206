## Unit Testing with Vitest

### Configuration

**`vitest.config.ts`:**
```typescript
import { defineConfig } from "vitest/config";

export default defineConfig({
  test: {
    globals: true,
    environment: "happy-dom",
```

### Example: Date Utility Tests

**`src/__tests__/utils/date.test.ts`:**
```typescript
import { describe, expect, it } from "vitest";
import { formatDate } from "../../utils/date";

describe("formatDate", () => {
  it("should format Date object to Japanese format", () => {
    const date = new Date("2024-01-15");
    const result = formatDate(date);
    expect(result).toBe("2024年1月15日");
  });

  it("should format string date to Japanese format", () => {
    const result = formatDate("2024-12-25");
    expect(result).toBe("2024年12月25日");
  });

  it("should accept custom format options", () => {
    const date = new Date("2024-06-01");
    const result = formatDate(date, {
      year: "numeric",
      month: "short",
    });
    expect(result).toBe("2024年6月");
  });

  it("should handle edge case dates", () => {
    // First day of year
    expect(formatDate(new Date("2024-01-01"))).toBe("2024年1月1日");
    // Last day of year
    expect(formatDate(new Date("2024-12-31"))).toBe("2024年12月31日");
  });
});
```

### Test Structure

**Import statements:**
```typescript
import { describe, expect, it } from "vitest";
import { functionToTest } from "../../utils/module";
```

**Test organization:**
```typescript
describe("functionName", () => {
  it("should do something", () => {
    // Arrange
    const input = "test";
    
    // Act
    const result = functionToTest(input);
    
    // Assert
    expect(result).toBe("expected");
  });
});
```

## E2E Testing with Playwright

### Configuration

**`playwright.config.ts`:**
```typescript
import { defineConfig, devices } from "@playwright/test";

export default defineConfig({
  testDir: "./tests",
  fullyParallel: true,
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 2 : 0,
  workers: process.env.CI ? 1 : undefined,
  reporter: "html",
  use: {
    baseURL: "http://localhost:4321",
    trace: "on-first-retry",
  },
  projects: [
    {
      name: "chromium",
      use: { ...devices["Desktop Chrome"] },
    },
    {
      name: "firefox",
      use: { ...devices["Desktop Firefox"] },
    },
    {
      name: "webkit",
      use: { ...devices["Desktop Safari"] },
    },
  ],
  webServer: {
    command: "pnpm dev",
    url: "http://localhost:4321",
    reuseExistingServer: !process.env.CI,
  },
});
```

### Example: Links Page Tests

**`tests/links.spec.ts`:**
```typescript
import { expect, test } from "@playwright/test";

test.describe("Links Page", () => {
  test("should have correct title and heading", async ({ page }) => {
    await page.goto("/links");

    await expect(page).toHaveTitle(/Links/);
    await expect(page.locator("h1").first()).toContainText("Links");
  });

  test("should display page description", async ({ page }) => {
    await page.goto("/links");

    await expect(
      page.locator("p").filter({ hasText: "SNS・ソーシャルメディア" }),
    ).toBeVisible();
  });

  test("should display SNS links", async ({ page }) => {
    await page.goto("/links");

    // GitHubリンクが存在することを確認
    await expect(
      page.locator('a[href*="github.com"]').first(),
    ).toBeVisible();
  });

  test("should have external links with correct attributes", async ({
    page,
  }) => {
    await page.goto("/links");

    // 外部リンクが存在することを確認
    const externalLinks = page.locator('a[href^="https://"]');
    await expect(externalLinks.first()).toBeVisible();
  });
});
```

### Test Patterns

**Navigation:**
```typescript
await page.goto("/path");
```

**Assertions:**
```typescript
// Page title
await expect(page).toHaveTitle(/Title/);

// Element visibility
await expect(page.locator("h1")).toBeVisible();

// Text content
await expect(page.locator("h1")).toContainText("Text");

// Element count
await expect(page.locator(".card")).toHaveCount(5);
```

**Selectors:**
```typescript
// CSS selector
page.locator(".class-name")

// Text filter
page.locator("p").filter({ hasText: "specific text" })

// Attribute selector
page.locator('a[href*="github.com"]')

// First/last
page.locator("button").first()
page.locator("button").last()
```

### CI Configuration

In CI environments:
- Tests retry up to 2 times on failure
- Single worker (no parallelization)
- Fails if `test.only` is found
- Doesn't reuse existing dev server

## Test Organization

### Unit Test Structure

```typescript
// src/__tests__/utils/example.test.ts
import { describe, expect, it } from "vitest";
import { example } from "../../utils/example";

describe("Module Name", () => {
  describe("functionName", () => {
    it("should handle normal case", () => {
      expect(example("input")).toBe("output");
    });
    
    it("should handle edge case", () => {
      expect(example("")).toBe("");
    });
    
    it("should throw on invalid input", () => {
      expect(() => example(null)).toThrow();
    });
  });
});
```

### E2E Test Structure

```typescript
// tests/feature.spec.ts
import { expect, test } from "@playwright/test";

test.describe("Feature Name", () => {
  test("should do something", async ({ page }) => {
    await page.goto("/path");
    await expect(page.locator(".element")).toBeVisible();
  });
  
  test("should handle interaction", async ({ page }) => {
    await page.goto("/path");
    await page.click("button");
    await expect(page.locator(".result")).toContainText("Success");
  });
});
```


- Lightweight DOM implementation

## Best Practices

### Unit Tests

1. **Test Pure Functions** - Focus on utility functions
2. **Use Descriptive Names** - Clear "should" statements
3. **Arrange-Act-Assert** - Follow AAA pattern
4. **Test Edge Cases** - Empty strings, null, undefined
5. **Mock External Dependencies** - Isolate units

### E2E Tests

1. **Test User Flows** - Real user interactions
2. **Use Semantic Selectors** - Prefer text/role over CSS
3. **Wait for Elements** - Use Playwright's auto-waiting
4. **Test Across Browsers** - All three projects
5. **Keep Tests Independent** - No test dependencies

### General

1. **Write Tests First** - TDD when possible
2. **Keep Tests Fast** - Quick feedback loop
3. **Maintain Tests** - Update with code changes
4. **Use CI** - Run tests automatically
5. **Monitor Coverage** - Aim for high coverage


### Playwright Debugging

```bash
# Debug mode
pnpm playwright test --debug

# UI mode
pnpm test:e2e:ui

# Headed mode (see browser)
pnpm playwright test --headed

# Slow motion
pnpm playwright test --slow-mo=1000
```

### Trace Viewer

Playwright captures traces on first retry:

```bash
pnpm playwright show-trace trace.zip
```
